TERMINALS

<DECLARE_FREE> ::= "$" | "free"
<EXISTS> ::= "∃" | "?" | "some"
<FORALL> ::= "∀" | "°" | "all"
<EQUIV> ::= "↔" | "⇔" | "<->" | "<=>"
<IMPLY> ::= "→" | "⇒" | "->" | "=>"
<OR> ::= "∨" | "|" | "||"
<XOR> ::= "∨̇" | "xor" | "^"
<AND> ::= "∧" | "&" | "&&"
<NOT> ::= "¬" | "~" | "!"
<EQUAL> ::= "=" | "=="
<UNEQUAL> ::= "≠" | "!="
<LESS> ::= "<"
<GREATER> ::= ">"
<LESS_EQUAL> ::= "≤" | "=<"
<GREATER_EQUAL> ::= "≥" | ">="
<LAMBDA> ::= "λ" | "\\"
