TERMINALS

<DECLARE_FREE> ::= "$" | "free"
<EQUIV> ::= "↔" | "⇔" | "<->" | "<=>"
<IMPLY> ::= "→" | "⇒" | "⊃" | "->" | "=>"
<OR> ::= "∨" | "|" | "||"
<XOR> ::= "∨̇" | "xor" | "^"
<AND> ::= "∧" | "&" | "&&"
<NOT> ::= "¬" | "~" | "!"
<EXISTS> ::= "∃" | "?" | "some"
<FORALL> ::= "∀" | "Â°" | "all"
<BOX> ::= "□" | "[]"
<DIAMOND> ::= "◇" | "<>"
<EQUAL> ::= "=" | "=="
<UNEQUAL> ::= "≠" | "!="
<LESS> ::= "<"
<GREATER> ::= ">"
<LESS_EQUAL> ::= "≤" | "=<"
<GREATER_EQUAL> ::= "≥" | ">="
<PLUS> ::= "+"
<MINUS> ::= "-"
<TIMES> ::= "⋅" | "*"
<DIVIDE> ::= "∕" | "/"
<LAMBDA> ::= "λ" | "\"
<PI> ::= "Π" | "\\"
<IDENTIFIER> ::= ["A"-"Z","_"] ["A"-"Z","_","0"-"9"]*
<INTEGER_LITERAL> ::= ["0"-"9"]+
<FLOATING_POINT_LITERAL> ::= ["0"-"9"]+ "." ["0"-"9"]+ (["e","E"] ["+","-"]? ["0"-"9"]+)?
<STRING_LITERAL> ::= "\"" ~["\"","\n",...]* "\""

Quantifiers and modal operators bind strong. The precedence order is thus

↔,⊃,∨,∨̇,∧,¬,∃,∀,□,◇,=,≠,<,≤,>,≥
where ∀ binds stronger than ∧ etc.
